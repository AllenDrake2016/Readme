    function AddNoteForLocation() {
        var txtSourceId = 'pgDInfo_dForfeiture_tbLocationNotes_I';
        var value = $("#" + txtSourceId).val();

        $.ajax({
            type: 'POST',
            url: '/WebMethodService.aspx/BuildLocationNote',
            data: JSON.stringify({ note: value }),
            contentType: 'application/json; charset=utf-8',
            dataType: 'json',
            success: function (msg) {
                $("#pgDInfo_dForfeiture_tmpLocationNote").val($('<div/>').text(msg.d).html());
                $("#divLocationNotesCurrent").html("<div id=\'tbLocationNotesCurrent\'>Location Notes: " + msg.d + "</div>");

                $("#" + txtSourceId).prop('disabled', true);
                $("#" + txtSourceId).attr('readonly', true);
                $("#" + txtSourceId).val('');
                DxEnable(_btnAddLocationNotesCollection);
            }
        });
    }
